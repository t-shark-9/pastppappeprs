                                                                                               ‫الشعر‪:‬‬

‫اإلجبثبد انًزٕسطخ إنى انجٍذح‪ :‬رقٕو ثًُبقشخ دٔس انشيض ٔأًٍْزّ فً عًهٍخ انزعجٍش عٍ أفكبس انشبعش‪.‬‬     ‫‪.1‬‬
‫انشجٕع إنى أعًبل يذسٔسخ يٍ أجم االسزشٓبد أيش ْبو‪ ،‬ثبإلظبفخ إنى ركش ثعط انًاليخ األدثٍخ انزً‬
                                                                      ‫رسبعذ فً إثشاص ْزا انذٔس‪.‬‬
                                 ‫ُ‬
‫اإلجبثبد انجٍذح إنى انًًزبصح‪ :‬رُبقش ثصٕسح أكثش عًقب ٔأشذ رشكٍضا ٔأًٍْخ انشيض ٔدٔسِ فً انزعجٍش عٍ‬
‫غشض انقصٍذح ٔأفكبس انشبعش يٍ خالل أعًبل أدثٍخ يذسٔسخ‪ٌٔ .‬جت انزشكٍض أٌعب ً عهى انًاليخ‬
                             ‫األدثٍخ فً ْزِ األعًبل نجٍبٌ أًٍْخ دٔس انشيض ٔعشظّ ثصٕسح ٔاظذخ‪.‬‬


‫اإلجبثبد انًزٕسطخ إنى انجٍذح‪ :‬رُبقش أًٍْخ انذٔس انزي ٌقٕو ثّ انشبعش فً عًهٍخ اإلصالح األجزًبعً‪،‬‬    ‫‪.2‬‬
‫يٍ خالل أعًال أدثٍخ يذسٔسخ‪ .‬كًب ٌجت إثشاص ثعط انًاليخ األدثٍخ انزً رسبعذ فً إثشاص ْزِ انعًهٍخ‬
                                                                              ‫ٔأثشْب فً انًجزًع‪.‬‬
‫اإلجبثبد انجٍذح إنى انًًزبصح‪ :‬رقذو يُبقشخ أكثش عًقب ً ٔرشكٍضاً نذٔس انشبعش فً عًهٍخ اإلصالح‬
‫اإلجزًبعً‪ ،‬يٍ خالل انشجٕع إنى أعًبل أدثٍخ يذسٔسخ يٍ أجم إسزشٓبد ٔثٍبٌ أًٍْخ ْزا انذٔس فً‬
                 ‫انًجزًع‪ .‬انزشكٍض عهى انًاليخ األدثٍخ ظشٔسي إلسُبد اإلجبثخ ٔإثشاصْب ثصٕسح يقُعخ‪.‬‬

                                                                                                  ‫الرواية‪:‬‬

‫اإلجبثبد انًزٕسطخ إنى انجٍذح‪ :‬رقٕو يُبقشخ أًٍْخ عبغفخ انعغت ٔدٔسْب فً انزأثٍش عهى رصشفبد ثطم‬           ‫‪.3‬‬
‫انشٔاي‪ ،‬يٍ خالل انشجٕع إنى أعًبل أدثٍخ يذسٔسخ راد صهخ ثبنًٕظٕع نجٍبٌ أًٍْخ انذٔس انزي رقٕو ثّ‬
                                                               ‫ْزِ انعبغفخ فً رطٕس أدذاس انشٔاٌخ‪.‬‬
‫اإلجبثبد انجٍذح إنى انًًزبصح‪ :‬رُبقش ثصٕسح يفصهخ ٔيعًقخ دٔس عبغفخ انغعت ٔاًٍْزٓب فً سٍش‬
‫أدذاس انشٔاٌخ ٔرطٕسْب‪ .‬انشجٕع إنى أعًبل أدثٍخ يذسٔسخ أيش ْبو فً اإلجبثخ نجٍبٌ أًٍْخ انذٔس انزي‬
                                                              ‫رقٕو ثّ ْزِ انعبغفخ فً انعًم انشٔأي‪.‬‬


‫اإلجبثبد انًزٕسطخ إنى انجٍذح‪ :‬رُبقش أًٍْخ انذٔس انزي رقٕو ثّ انًشأح فً انزأثٍش عهى أدذاس انشٔاٌخ‬       ‫‪.4‬‬
‫ٔدٔسْب فً رطٕس انعًم انشٔائً‪ ،‬يٍ خالل انشجٕع إنى أعًبل سٔائٍخ يذسٔسخ نجٍبٌ انًشأح انٓبو فً‬
                                                                                           ‫انشٔاٌخ‪.‬‬
‫اإلجبثبد انجٍذح إنى انًًزبصح‪ :‬رقذو يُبقشخ أكثش عًقب ً ٔأشذ رشكٍضاً ٔدٔس انًشأح انٓبو فً انزأثٍش عهى‬
 ‫أدذاس انشٔاٌخ‪ .‬االسزشٓبد ثأعًبل سٔائٍخ راد صهخ ثٓزا انًٕظٕع أيش أسبسً نجٍبٌ ْزا انذٔس ٔأًٍْزّ‪.‬‬



                                                                                          ‫القصة القصيرة‪:‬‬

‫اإلجبثبد انًزٕسطخ إنى انجٍذح‪ :‬رُبقش أًٍْخ انذبنخ انُفسٍخ نهكبرت ٔأًٍْزٓب فً َجبح عًهّ انقصص‪.‬‬           ‫‪.5‬‬
‫انشجٕع إنى ثعط انًجًٕعبد انقصصٍخ انزً رًذ دساسزٓب أيش ْبو السزشٓبد ٔثٍبٌ أًٍْخ دٔس رجشثخ‬
                                                              ‫انكبرت األدثٍخ فً َجبح انقصخ انقصٍشح‪.‬‬
‫اإلجبثبد انجٍذح إنى انًًزبصح‪ :‬رُبقش ثصٕسح أكثش عًقب ً ٔرفصٍالً أًٍْخ انذبنخ انُفسٍخ نهكبرت ٔدٔسْب فً‬
‫َجبح قصصّ انقصٍشح يٍ خالل انشجٕع إنى يجًٕعبد قصصٍخ رًذ دساسزٓب نغبٌخ االسزشٓبد ٔدعى‬
                                                                                            ‫اإلجبثخ‪.‬‬


‫اإلجبثبد انًزٕسطخ إنى انجٍذح‪ :‬رُبقش دٔس انجٍئخ انًذهٍخ فً ثُبء انقصخ انقصٍشح ٔأًٍْزٓب فً سسى‬           ‫‪.6‬‬
‫أدذاثٓب‪ٔ .‬انشجٕع إنى ثعط انًجًٕعبد انقصصٍخ انزً رًذ دساسزٓب أيش ْبو نزأٌٍذ اإلجبثخ ٔإسُبدْب‬
                                                                                    ‫ثشٕاْذ عًهٍخ‪.‬‬
                                ‫ً‬           ‫ً‬
‫اإلجبثبد انجٍذح إنى انًًزبصح‪ :‬رعشض يُبقشخ أكثش رفصٍال ٔأشذ رشكٍضا نهذٔس انٓبو انزي رفٕو ثّ انجٍئخ‬
‫انًذهٍخ فً سسى األدذاس فً انقصخ انقصٍشح‪ٌٔ .‬جت انشجٕع إنى يجًٕعبد قصصٍخ يذسٔسخ نجٍبٌ‬
                                              ‫أًٍْخ ْزا انذٔس ٔرأثٍشِ فً عًهٍخ رٕظٍخ انعًم انقصصً‪.‬‬

                                                                                              ‫المسرح‪:‬‬

‫اإلجبثبد انًزٕسطخ إنى انجٍذح‪ :‬رُبقش يٕظٕع انخهفٍخ انثقبفٍخ ٔأًٍْخ دٔسِ فً انعًم انًسشدً يٍ‬          ‫‪.7‬‬
‫خالل انشجٕع عهى األقم إنى يسشدٍزٍٍ يذسٔسزٍٍ نجٍبٌ أًٍْخ ْزا انذٔس‪ .‬إٌ ثٍبٌ ثعط انًاليخ األدثٍخ‬
                                                          ‫أيش ظشٔسي إلثشاص انًٕظٕع ٔرٕظٍذّ‪.‬‬
‫اإلجبثبد انجٍذح إنى انًًزبصح‪ :‬رُبقش ثزفصٍم أكثش ٔرشكٍض أكجش يٕظٕع انخهفٍخ انثقبفٍخ ٔأًٍْخ دٔسِ فً‬
‫انًسشدٍخ‪ .‬االسزشٓبد ثجعط األعًبل انًسشدٍخ انًذسٔسخ أيش ْبو فً انًُبقشخ نهزٕظٍخ‪ .‬االسزشٓبد‬
                                  ‫كًب ٌجت عشض انًاليخ األدثٍخ انزً رسبعذ فً إثشاص انذٔس ٔأًٍْزّ‪.‬‬


‫اإلجبثبد انًزٕسطخ إنى انجٍذح‪ :‬رُبقش أًٍْخ انذٔس انزي رقٕو ثّ انًٕسٍقى انًصبدجخ نهعًم انًسشدً‬        ‫‪.8‬‬
‫ٔرأثٍشْب فً إثشاص انعًم انًسشدً ٔعشظّ أيبو انجًٕٓس‪ .‬االسزشٓبد ثأعًبل يسشدٍخ يذسٔسخ أيش‬
                       ‫ظشٔسي نجٍبٌ دٔس انًٕسٍقى انًصبدجخ ٔأًٍْزٓب فً عشض األعًبل انًسشدٍخ‪.‬‬
‫اإلجبثبد انجٍذح إنى انًًزبصح‪ :‬رُبقش ثصٕسح يفصهخ ٔيشكضح انذٔس انٓبو انزي رقٕو ثّ انًٕسٍقى‬
‫انًصبدجخ نهعًم انًسشدً‪ .‬االسزشٓبد ثأعًبل يسشدٍخ يذسٔسخ أيش ظشٔسي نجٍبٌ ْزا انذٔس ٔأًٍْزّ‬
                                                                 ‫فً عشض األعًبل انًسشدٍخ‪.‬‬

                                                                                                       ‫عام‪:‬‬

‫اإلجبثبد انًزٕسطخ إنى انجٍذح‪ :‬رُبقش أًٍْخ انعالقخ ثٍٍ األدة ٔانذٍبح ٔدٔس ْزِ انعالقخ فً انزأثٍش عهى‬      ‫‪.9‬‬
            ‫األفشاد‪ .‬االسزشٓبد ثأعًبل أدثٍخ يذسٔسخ أيش ظشٔسي نجٍبٌ أًٍْخ ْزِ انعالقخ فً انًجزًع‪.‬‬
                         ‫كًب ٌزٕقع إثشاص دٔس األدٌت فً ْزِ انعالقخ نزٕظٍخ رأثٍشِ انفكشي عهى األفشاد‪.‬‬
‫اإلجبثبد انجٍذح إنى انًًزبصح‪ :‬رُبقش ثصٕسح أكجش رفصٍالً ٔأكثش عًقب أًٍْخ انعالقخ ثٍٍ األدة ٔانذٍبح‬
‫ٔدٔسْب فً انزأثٍش عهى األفشاد‪ .‬االسزشٓبد ثأعًبل أدثٍخ يذسٔسخ أيش ْبو نجٍبٌ أًٍْخ ْزِ انعالقخ ٔدٔس‬
                                        ‫األدٌت فٍٓب إلثشاص دٔسِ انفكشي فً انزأثٍش عهى أفشاد انًجزًع‪.‬‬


‫‪ .10‬اإلجبثبد انًزٕسطخ إنى انجٍذح‪ :‬رقذو يُبقشخ أًٍْخ األدة انٕاقعً ٔدٔسِ فً إثشاص انقعبٌب األدثٍخ‬
‫ٔاالجزًبعٍخ ٔاإلَسبٍَخ‪ .‬االسزشٓبد ثأعًبل أدثٍخ يذسٔسخ أيش ظشٔسي نجٍبٌ أًٍْخ ْزا انذٔس‪ .‬كًب‬
       ‫ٌزٕقع عشض ثعط انجٕاَت انفٍُخ انزً ٌٕظفٓب األدٌت فً أعًبنّ إلثشاص انجبَت انٕاقعً فً األدة‬
‫اإلجبثبد انجٍذح إنى انًًزبصح‪ :‬رُبقش ثصٕسح يعًقخ ٔيفصهخ أًٍْخ األدة انٕاقعً ٔدٔسِ فً إثشاص‬
‫انقعبٌب األدثٍخ ٔاالجزًبعٍخ ٔاإلَسبٍَخ‪ .‬عشض األعًبل األدثٍخ انًذسٔسخ ايش ْبو نجٍبٌ دقٍقخ ْزا انذٔس‬
  ‫ٔأًٍْزّ‪ .‬أيب عشض انجٕاَت انفٍُخ انزً ٌٕظفٓب األدٌت فً أعًبنّ فأي ٌش ْبو نجٍبٌ أًٍْخ األدة انٕاقعً‪.‬‬


‫‪ .11‬اإلجبثبد انًزٕسطخ إنى انجٍذح‪ٌ :‬زٕقع أٌ رُبقش دٔس عُصش انزشٌٕق ٔاًٍْزّ فً انزعجٍش عٍ انقعبٌب‬
‫األدثٍخ‪ .‬االسزشٓبد ثأعًبل أدثٍخ يذسٔسخ أيش ظشٔسي نجٍبٌ أًٍْخ ْزا انعُصش‪ ،‬يع عشض نجعط‬
                                                                ‫انًاليخ األدثٍخ انزً رسبْى فً إثشاصِ‪.‬‬
‫اإلجبثبد انجٍذح إنى انًًزبصح‪ :‬رقذو يُبقشخ أكثش عًقب ً ٔرفصٍالً نهذٔس انزي ٌقٕو ثّ عُصش انزشٌٕق فً‬
‫األعًبل األدثٍخ ٔأًٍْزّ فً إثشاص انقعبٌب األدثٍخ‪ .‬االسزشٓبد ثأعًبل أدثٍخ يذسٔسخ أيش ْبو إلثشاص أًٍْخ‬
‫ْزا انذٔس‪ .‬إٌ يُبقشخ انًاليخ األدثٍخ انزً رسبْى فً ثٍبٌ ْزا انعُصش أيش ظشٔسي نزٕظٍخ انذٔس انزي‬
                                                              ‫ٌقٕو ثّ ْزا انعُصش فً األعًبل األدثٍخ‪.‬‬


‫‪ .12‬اإلجبثبد انًزٕسطخ إنى انجٍذح‪ :‬رُبقش دٔس عالقخ انعُٕاٌ ثبنعًم األدثً ٔأًٍْزّ فً إثشاص فكشح انكبرت‪.‬‬
‫كًب ٌجت االسزشٓبد ثعًهٍٍ أدثٍٍٍ رًذ دساسزًٓب نزٕظٍخ أًٍْخ انذٔس انزي ٌقٕو ثّ انعُٕاٌ فً رٕظٍخ‬
                                                                                         ‫فكشح انكبرت‪.‬‬
                                         ‫ً‬
‫اإلجبثبد انجٍذح إنى انًًزبصح‪ :‬رقذو يُبقشخ يعًقخ ٔأكثش رفصٍال نهعالقخ ثٍٍ انعُٕاٌ ٔانعًم األدثً ٔأًٍْخ‬
‫ْزِ انعالقخ فً انعجٍش عٍ فكشح انكبرت‪ .‬انشجٕع إنى أعًبل أدثٍخ يذسٔسخ أيش ظشٔسي االسزشٓبد‬
